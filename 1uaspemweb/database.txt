#buat database di phpmyadmin

CREATE DATABASE IF NOT EXISTS uaspemweb;

USE uaspemweb;

CREATE TABLE IF NOT EXISTS jurusan (
    kode_jurusan VARCHAR(10) PRIMARY KEY,
    nama_jurusan VARCHAR(100) NOT NULL
);

CREATE TABLE IF NOT EXISTS mata_kuliah (
    id INT AUTO_INCREMENT PRIMARY KEY,
    kode_jurusan VARCHAR(10),
    kode_mata_kuliah VARCHAR(10) NOT NULL,
    mata_kuliah VARCHAR(100) NOT NULL,
    semester INT NOT NULL,
    FOREIGN KEY (kode_jurusan) REFERENCES jurusan(kode_jurusan)
);

CREATE TABLE IF NOT EXISTS dosen (
    kode_dosen VARCHAR(10) PRIMARY KEY,
    nama_dosen VARCHAR(100) NOT NULL
);

CREATE TABLE IF NOT EXISTS mahasiswa (
    nim VARCHAR(15) PRIMARY KEY,
    nama_mahasiswa VARCHAR(100) NOT NULL,
    kode_jurusan VARCHAR(10),
    FOREIGN KEY (kode_jurusan) REFERENCES jurusan(kode_jurusan)
);
